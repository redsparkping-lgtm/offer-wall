function renderData(data) { const container = document.getElementById("offers"); // Group by category const grouped = {}; data.forEach(offer => { const category = offer.category || "Other"; if (!grouped[category]) { grouped[category] = []; } grouped[category].push(offer); }); let html = ""; // Loop categories for (let category in grouped) { html += `
🔥 ${category}
`; // Loop offers inside category grouped[category].forEach(offer => { html += `
${offer.name}
Offer ID
${offer.id}
Payout
$${offer.payout}
GEO
${offer.geo}
Allowed Traffic
${offer.traffic}
Preview
View Offer
`; }); html += `
`; } container.innerHTML = html; }
